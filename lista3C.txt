        if(v[i] > max){
            max = v[i];
        }
    }
    int v2[max][n];
    for(j = 0; j < n; j++){
        for(i = 0; i < max; i++){
            if(i < max - v[j]){
                v2[i][j] = 0;
            } else {
                v2[i][j] = 1;
            }
        }
    }
    for(i = 0; i < max; i++){
        for(j = 0; j < n; j++){

                printf("*");
            }

        if(frase[0] == 32){
            printf("A frase digitada nao tem nenhuma palavra\n");
        } else{
            printf("A frase digitada tem 1 palavra\n");
        }
    } else if(tam == 2){
        if(frase[0] == 32){
            if(frase[1] != 32){
                printf("A frase digitada tem 1 palavra\n");
            } else {
                printf("A frase digitada nao tem nenhuma palavra\n");
            }
        } else {
            if(frase[1] != 32){
                printf("A frase digitada tem 2 palavras\n");
            } else {
                printf("A frase digitada tem 1 palavra\n");
            }
        }
    } else if(tam >= 3){
        for(i = 1; i < tam; i++){
            if(frase[i] == 32){
                if(frase[i - 1] != 32){
                    verf++;
                }
            }
            if(i == tam - 1){
                if(frase[i] != 32){
                    verf++;
                }
            }
        }
        printf("A frase tem %d palavras\n", verf);
